209 #ifdef HAL_IRRX_DESIGN_LEGACY
210  /*
211  * - Use IRRX in the PWD mode. \n
212  * - Step1: Call #hal_gpio_init() to init the pins, if EPT tool hasn't been used to configure the related pinmux.
213  * - Step2: Set GPIO34 to RX function to enable the IRRX mode.
214  * - Step3: Call hal_irrx_init() to initialize the IRRX.
215  * - Step4: Set the PWD parameters: inverse and terminate_threshold.
216  * - Step5: Call hal_irrx_receive_pwd_start() to start the hardware.
217  * - Step6: Call hal_irrx_receive_pwd() to receive the PWD code.
218  * - Step7: Call hal_irrx_receive_pwd_start() once again to restart the PWD mode.
219  * - Step8: Call hal_irrx_deinit() to deinitialize the IRRX.
220  * - sample code:
221  * @code
222  * //Init GPIO, set GPIO pinmux(if EPT tool hasn't been used to configure the related pinmux).
223  * hal_gpio_init(HAL_GPIO_34);
224  *
225  * //Call hal_pinmux_set_function() to set GPIO pinmux, for more information, please reference hal_pinmux_define.h
226  * //function_index = HAL_GPIO_34_IR_RX.Need not to configure pinmux if EPT tool is used.
227  * hal_pinmux_set_function(HAL_GPIO_34, function_index);//Set the GPIO34 to IRRX mode.
228  *
229  * hal_irrx_init();
230  * uint32_t us = 0;
231  * hal_irrx_pwd_config_t format = {0}; //Initializes the PWD parameter.
232  * format.inverse = 1;
233  * format.terminate_threshold = 10200;
234  * if (HAL_IRRX_STATUS_OK != hal_irrx_receive_pwd_start(&format,
235  * user_callback,
236  * NULL,
237  * &us)) {
238  * dbg_error("hal_irrx_receive_pwd_start fail!");
239  * }
240  * if (HAL_IRRX_STATUS_OK != hal_irrx_receive_pwd(&received_length,
241  * data,
242  * sizeof(data))) {
243  * dbg_error("hal_irrx_receive_pwd fail!");
244  * }
245  * @endcode
246  * @code
247  * // Callback function. This function should be registered with #hal_irrx_receive_pwd().
248  * void user_callback (void *user_data)
249  * {
250  * hal_irrx_deinit();
251  * // user's handler
252  * }
253  *
254  * @endcode
255  *
256  *
257  * - Use IRRX driver to trigger IRRX RC5/PWD manually. \n
258  *
259  *
260  */
261 #endif
